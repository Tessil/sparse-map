90 // From Wikipedia: https://en.wikipedia.org/wiki/Hamming_weight
91 inline int fallback_popcountll(unsigned long long int x) {
92  static_assert(sizeof(unsigned long long int) == sizeof(std::uint64_t),
93  "sizeof(unsigned long long int) must be equal to sizeof(std::uint64_t). "
94  "Open a feature request if you need support for a platform where it isn't the case.");
95 
96  const std::uint64_t m1 = 0x5555555555555555ull;
97  const std::uint64_t m2 = 0x3333333333333333ull;
98  const std::uint64_t m4 = 0x0f0f0f0f0f0f0f0full;
99  const std::uint64_t h01 = 0x0101010101010101ull;
100 
101  x -= (x >> 1ull) & m1;
102  x = (x & m2) + ((x >> 2ull) & m2);
103  x = (x + (x >> 4ull)) & m4;
104  return static_cast<int>((x * h01) >> (64ull - 8ull));
105 }

107 inline int fallback_popcount(unsigned int x) {
108  static_assert(sizeof(unsigned int) == sizeof(std::uint32_t) || sizeof(unsigned int) == sizeof(std::uint64_t),
109  "sizeof(unsigned int) must be equal to sizeof(std::uint32_t) or sizeof(std::uint64_t). "
110  "Open a feature request if you need support for a platform where it isn't the case.");
111 
112  if (sizeof(unsigned int) == sizeof(std::uint32_t)) {
113  const std::uint32_t m1 = 0x55555555;
114  const std::uint32_t m2 = 0x33333333;
115  const std::uint32_t m4 = 0x0f0f0f0f;
116  const std::uint32_t h01 = 0x01010101;
117 
118  x -= (x >> 1) & m1;
119  x = (x & m2) + ((x >> 2) & m2);
120  x = (x + (x >> 4)) & m4;
121  return static_cast<int>((x * h01) >> (32 - 8));
122  }
123  else {
124  return fallback_popcountll(x);
125  }
126 }

225 inline std::size_t round_up_to_power_of_two(std::size_t value) {
226  if(is_power_of_two(value)) {
227  return value;
228  }
229 
230  if(value == 0) {
231  return 1;
232  }
233 
234  --value;
235  for(std::size_t i = 1; i < sizeof(std::size_t) * CHAR_BIT; i *= 2) {
236  value |= value >> i;
237  }
238 
239  return value + 1;

243 /**
244  * WARNING: the sparse_array class doesn't free the ressources allocated through the allocator passed in parameter
245  * in each method. You have to manually call `clear(Allocator&)` when you don't need a sparse_array object anymore.
246  *
247  * The reason is that the sparse_array doesn't store the allocator to avoid wasting space in each sparse_array when
248  * the allocator has a size > 0. It only allocates/deallocates objects with the allocator that is passed in parameter.
249  *
250  *
251  *
252  * Index denotes a value between [0, BITMAP_NB_BITS), it is an index similar to std::vector.
253  * Offset denotes the real position in `m_values` corresponding to an index.
254  *
255  * We are using raw pointers instead of std::vector to avoid loosing 2*sizeof(size_t) bytes to store the capacity
256  * and size of the vector in each sparse_array. We know we can only store up to BITMAP_NB_BITS elements in the array,
257  * we don't need such big types.
258  *
259  *
260  * T must be nothrow move contructible and/or copy constructible.
261  * Behaviour is undefined if the destructor of T throws an exception.
262  *
263  * See https://smerity.com/articles/2015/google_sparsehash.html for details on the idea behinds the implementation.
264  *
265  * TODO Check to use std::realloc and std::memmove when possible
266  */

312  /**
313  * Map an ibucket [0, bucket_count) in the hash table to a sparse_ibucket
314  * (a sparse_array holds multiple buckets, so there is less sparse_array than bucket_count).
315  *
316  * The bucket ibucket is in m_sparse_buckets[sparse_ibucket(ibucket)][index_in_sparse_bucket(ibucket)]
317  * instead of something like m_buckets[ibucket] in a classical hash table.
318  */
319  static std::size_t sparse_ibucket(std::size_t ibucket) {
320  return ibucket >> BUCKET_SHIFT;
321  }
322 
323  /**
324  * Map an ibucket [0, bucket_count) in the hash table to an index in the sparse_array
325  * which corresponds to the bucket.
326  *
327  * The bucket ibucket is in m_sparse_buckets[sparse_ibucket(ibucket)][index_in_sparse_bucket(ibucket)]
328  * instead of something like m_buckets[ibucket] in a classical hash table.
329  */
330  static typename sparse_array::size_type index_in_sparse_bucket(std::size_t ibucket) {
331  return static_cast<typename sparse_array::size_type>(ibucket & sparse_array::BUCKET_MASK);

593  /**
594  * Insertion
595  *
596  * Two situations:
597  * - Either we are in a situation where std::is_nothrow_move_constructible<value_type>::value is true.
598  * In this case, on insertion we just reallocate m_values when we reach
599  * its capacity (i.e. m_nb_elements == m_capacity), otherwise we just put the new value at its
600  * appropriate place. We can easly keep the strong exception guarantee as moving the values around is safe.
601  * - Otherwise we are in a situation where std::is_nothrow_move_constructible<value_type>::value is false.
602  * In this case on EACH insertion we allocate a new area of m_nb_elements + 1 where we copy the values
603  * of m_values into it and put the new value there. On success, we set m_values to this new area.
604  * Even if slower, it's the only way to preserve to strong exception guarantee.
605  */
606  template<typename... Args, typename U = value_type,
607  typename std::enable_if<std::is_nothrow_move_constructible<U>::value>::type* = nullptr>
608  void insert_at_offset(allocator_type& alloc, size_type offset, Args&&... value_args) {
609  if(m_nb_elements < m_capacity) {
610  insert_at_offset_no_realloc(alloc, offset, std::forward<Args>(value_args)...);
611  }
612  else {
613  insert_at_offset_realloc(alloc, offset, next_capacity(), std::forward<Args>(value_args)...);
614  }
615  }

617  template<typename... Args, typename U = value_type,
618  typename std::enable_if<!std::is_nothrow_move_constructible<U>::value>::type* = nullptr>
619  void insert_at_offset(allocator_type& alloc, size_type offset, Args&&... value_args) {

719  * - Either we are in a situation where std::is_nothrow_move_constructible<value_type>::value is true.

721  * - Otherwise we are in a situation where std::is_nothrow_move_constructible<value_type>::value is false.

723  * to this new area. Even if slower, it's the only way to preserve to strong exception guarantee.

1578  /**
1579  * We must insert the value in the first empty or deleted bucket we find. If we first find a
1580  * deleted bucket, we still have to continue the search until we find an empty bucket or until we have
1581  * searched all the buckets to be sure that the value is not in the hash table.
1582  * We thus remember the position, if any, of the first deleted bucket we have encountered
1583  * so we can insert it there if needed.
1584  */
1585  bool found_first_deleted_bucket = false;
1586  std::size_t sparse_ibucket_first_deleted = 0;
1587  typename sparse_array::size_type index_in_sparse_bucket_first_deleted = 0;
1588 
1589 
1590 
1591  const std::size_t hash = hash_key(key);
1592  std::size_t ibucket = bucket_for_hash(hash);
1593 
1594  std::size_t probe = 0;
1595  while(true) {
1596  std::size_t sparse_ibucket = sparse_array::sparse_ibucket(ibucket);
1597  auto index_in_sparse_bucket = sparse_array::index_in_sparse_bucket(ibucket);
1598 
1599  if(m_sparse_buckets[sparse_ibucket].has_value(index_in_sparse_bucket)) {
1600  auto value_it = m_sparse_buckets[sparse_ibucket].value(index_in_sparse_bucket);
1601  if(compare_keys(key, KeySelect()(*value_it))) {
1602  return std::make_pair(iterator(m_sparse_buckets.begin() + sparse_ibucket, value_it), false);
1603  }
1604  }
1605  else if(m_sparse_buckets[sparse_ibucket].has_deleted_value(index_in_sparse_bucket) &&
1606  probe < m_bucket_count)
1607  {
1608  if(!found_first_deleted_bucket) {
1609  found_first_deleted_bucket = true;
1610  sparse_ibucket_first_deleted = sparse_ibucket;
1611  index_in_sparse_bucket_first_deleted = index_in_sparse_bucket;
1612  }
1613  }
1614  else if(found_first_deleted_bucket) {
1615  auto it = insert_in_bucket(sparse_ibucket_first_deleted, index_in_sparse_bucket_first_deleted,
1616  std::forward<Args>(value_type_args)...);
1617  m_nb_deleted_buckets--;

1620  }
1621  else {

1625
